## Install

```bash
npm install --save hawkui
```

## Usage

```jsx
import React, { Component } from 'react'

import {Button} from 'hawkui'
import 'hawkui/dist/index.css'

class Example extends Component {
  render() {
    return <Button type="primary">Primary Button</Button>
  }
}
```
